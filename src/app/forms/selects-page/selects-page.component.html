@Input() autofocus: boolean Whether the component is auto-focused.
@Input() autosize: boolean Whether the component is autosized.

@Input() filterPlaceholder: string = 'Filter options...' Placeholder to display for the component's filter text input.
@Input() forceDownwardOpening: boolean Whether to force the component to open beneath itself.
@Input() groupBy: string NEEDS DOCUMENTATION.

@Input() identifier: string NEEDS DOCUMENTATION.
@Input() label: string The label for the component.

@Input() multiple: boolean Whether multiple selection is allowed.
@Input() name: string The name for the component.
@Input() options: SelectDropdownOption[] = [] The options to display for the component.
@Input() placeholder: string The placeholder for the component.
@Input() required: boolean Whether the user's input is required.
@Input() requiredIndicator: string | boolean = '*' The indicator displayed to inform the user that input is required.
@Input() selectCaret: string Accepts string or template reference. The select caret to display in place of the default caret, typically an icon.
@Input() tagging: boolean Whether selected options are displayed as tags.
