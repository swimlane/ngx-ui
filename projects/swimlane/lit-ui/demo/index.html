Select Usage

Import and use the select component in your application: 

<script type="module">
  import '@swimlane/lit-ui/select';
</script>

<script>
  const select = document.querySelector('swim-select');
  select.options = [
    { name: 'Option 1', value: 'opt1' },
    { name: 'Option 2', value: 'opt2' },
    { name: 'Option 3', value: 'opt3' }
  ];
</script>

<swim-select label="Choose an option" placeholder="Select..."></swim-select>

Button Group Usage

Import and use the button group component in your application: 

<script type="module">
  import '@swimlane/lit-ui/button-group';
  import '@swimlane/lit-ui/button';
</script>

<swim-button-group button-group-style="primary">
  <swim-button type="button">Save</swim-button>
  <swim-button type="button">Cancel</swim-button>
</swim-button-group>

    Icon Usage

    Import the icon component and ensure the ngx-icon font CSS is loaded (e.g. from ngx-ui or this demo's assets): 

    <script type="module">
  import '@swimlane/lit-ui/icon';
</script>
<!-- Load icon font CSS in your app -->
<link rel="stylesheet" href="path/to/ngx-icons.css" />

<swim-icon font-icon="plus"></swim-icon>
<swim-icon font-icon='["square-filled", "x :text-red"]'></swim-icon>
<swim-icon font-icon="x" alt="Close"></swim-icon>
<!-- Inline SVG from URL (defaultPath + svg-src + .svg) -->
<swim-icon svg-src="chart-scatter" default-path="assets/svgs"></swim-icon>
