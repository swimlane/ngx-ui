Import the split components (and icon for the handle grip). Use area-basis, min-basis, max-basis on areas; handle-basis on the handle (default 15px).

<script type="module">
  import '@swimlane/lit-ui/icon';
  import '@swimlane/lit-ui/split';
</script>

<div class="split-example">
  <swim-split direction="row">
    <swim-split-area area-basis="1 1 50%">Left content</swim-split-area>
    <swim-split-handle></swim-split-handle>
    <swim-split-area area-basis="1 1 50%">Right content</swim-split-area>
  </swim-split>
</div>
