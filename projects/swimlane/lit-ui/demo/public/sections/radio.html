Import and use the radio components:

<script type="module">
  import '@swimlane/lit-ui/radio';
</script>

<!-- Standalone (parent controls state) -->
<swim-radio value="A" .checked="${selected === 'A'}" @change="${e => selected = e.detail}">Option A</swim-radio>

<!-- Group -->
<swim-radio-group name="choice" @change="${e => console.log(e.detail)}">
  <swim-radio value="one">One</swim-radio>
  <swim-radio value="two">Two</swim-radio>
  <swim-radio value="three">Three</swim-radio>
</swim-radio-group>
