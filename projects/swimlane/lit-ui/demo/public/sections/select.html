Select Component

Dropdown select with filtering and multi-select support

Basic

Required

<!-- Basic -->
<swim-select label="Attack Type" placeholder="Select..."></swim-select>

<!-- Required -->
<swim-select label="Attack Type" required placeholder="Please select..."></swim-select>

<script>
  const options = [
    { name: 'Breach', value: 'breach' },
    { name: 'DDOS', value: 'ddos' },
    { name: 'Physical', value: 'physical' }
  ];
  document.querySelector('swim-select').options = options;
</script>

Options can be defined inline as <swim-option> children instead of setting the .options property via JavaScript. 

<swim-select label="Attack Type" required filterable="false">
  <swim-option name="Breach" value="breach"></swim-option>
  <swim-option name="DDOS" value="ddos"></swim-option>
  <swim-option name="Physical" value="physical"></swim-option>
</swim-select>

Import and use the select component in your application:

<script type="module">
  import '@swimlane/lit-ui/select';
</script>

<!-- Option A: Declarative options via swim-option children -->
<swim-select label="Attack Type" required>
  <swim-option name="Breach" value="breach"></swim-option>
  <swim-option name="DDOS" value="ddos"></swim-option>
  <swim-option name="Physical" value="physical"></swim-option>
</swim-select>

<!-- Option B: Programmatic options via .options property -->
<script>
  const select = document.querySelector('swim-select');
  select.options = [
    { name: 'Option 1', value: 'opt1' },
    { name: 'Option 2', value: 'opt2' },
    { name: 'Option 3', value: 'opt3' }
  ];
</script>

Properties

AttributeTypeDefaultDescription
labelstring''Label text
placeholderstring'Select...'Placeholder when no value selected
hintstring''Hint text below the select
.optionsArray<{name, value}>[]Array of selectable options
multiplebooleanfalseEnable multi-select mode
filterablebooleantrueEnable type-to-filter
allow-clearbooleantrueShow clear button when value selected
appearance'legacy' | 'fill''legacy'Visual appearance style
size'sm' | 'md' | 'lg''sm'Select size
disabledbooleanfalseDisables the select
requiredbooleanfalseMarks as required
