Disabled: tooltip does not show. No caret: tooltip shows without the arrow.

Disabled | No caret

Import and use the tooltip component.

<script type="module">
  import '@swimlane/lit-ui/tooltip';
</script>

<swim-tooltip content="Phishing Attack" type="tooltip" placement="top">
  <a href="#">Top</a>
</swim-tooltip>

<!-- Custom content via slot -->
<swim-tooltip type="popover" placement="bottom">
  <swim-button variant="primary">Open</swim-button>
  <div slot="content">Custom content here.</div>
</swim-tooltip>
<swim-tooltip>
