Use openDrawer() to create drawers programmatically, similar to Angular DrawerService.

Open Left (imperative) Open Bottom (imperative) 
import { openDrawer } from '@swimlane/lit-ui';

openDrawer({
  direction: 'left',
  size: 70,
  content: `
    <div class="drawer-demo-toolbar">Alert Everyone!</div>
    <div class="drawer-demo-section">
      <h1>Attack Type: Malware</h1>
    </div>
  `
});
 
import '@swimlane/lit-ui/drawer';

// Declarative
<swim-drawer
  id="myDrawer"
  direction="left"
  size="70"
  close-on-outside-click
  is-root
  z-index="998"
>
  <p>Slot content</p>
</swim-drawer>

// Show: document.getElementById('myDrawer').show();
// Hide: document.getElementById('myDrawer').hide();
// Listen: myDrawer.addEventListener('close', () => { ... });

// Imperative
import { openDrawer } from '@swimlane/lit-ui';
const { close, drawer } = openDrawer({ direction: 'left', size: 70, content: '...' });
close(); // or drawer.hide();
