Options can be defined inline as <swim-option> children instead of setting the .options property via JavaScript. 

<swim-select label="Attack Type" required filterable="false">
  <swim-option name="Breach" value="breach"></swim-option>
  <swim-option name="DDOS" value="ddos"></swim-option>
  <swim-option name="Physical" value="physical"></swim-option>
</swim-select>

Set grouped on <swim-select> to show section headers from each option’s group or category. Omit grouped and the same data renders as a flat list (metadata is still on each option). With grouped and filtering on, typing a section name (e.g. “Pantry”) matches options in that section.

Type at least 2 characters to search public GitHub users via the Search users API. Results load after the default 500ms debounce on filter-change.

Selected: — 

<swim-select
  id="githubUserSelect"
  label="GitHub user"
  placeholder="Pick a user…"
  async-filter
  filter-min-length="2"
  dropdown-align="center"
  loading
></swim-select>

<script type="module">
  const el = document.getElementById('githubUserSelect');
  el.addEventListener('filter-change', async (e) => {
    const q = e.detail.query;
    if (!q) { el.options = []; el.loading = false; return; }
    el.loading = true;
    const res = await fetch(
      `https://api.github.com/search/users?q=${encodeURIComponent(q)}&per_page=15`,
      { headers: { Accept: 'application/vnd.github+json' } }
    );
    const data = await res.json();
    el.options = (data.items ?? []).map((u) => ({
      name: u.login,
      value: u.login,
      title: u.login,
      description: u.type === 'Organization' ? 'Organization' : 'User',
    }));
    el.loading = false;
  });
</script>
